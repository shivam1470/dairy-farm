# Railway Configuration for Backend Deployment

# Build Command
build: pnpm install && cd apps/backend && npx prisma generate && pnpm build

# Start Command
start: cd apps/backend && npx prisma migrate deploy && node dist/main

# Environment Variables Required:
# - DATABASE_URL (PostgreSQL connection string)
# - JWT_SECRET (random secret for JWT tokens)
# - NODE_ENV=production

# Port
PORT: 3001
